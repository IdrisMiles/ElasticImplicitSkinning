138 template <typename T>
139 T Texture3DCpu<T>::TrilinearInterpolate(const float _x, const float _y, const float _z)
140 {
141 
142  //[0:1]
143  glm::vec3 texSpace = glm::vec3(m_textureSpaceTransform*(glm::vec4(_x, _y, _z, 1.0f)));
144 
145  // Get data coords
146  float x = texSpace.x * m_dim;
147  float y = texSpace.y * m_dim;
148  float z = texSpace.z * m_dim;
149 
150  unsigned int x0 = floor(x);
151  unsigned int y0 = floor(y);
152  unsigned int z0 = floor(z);
153 
154  // Adjust for potential out of bounds
155  x0 = x0 >= m_dim ? m_dim - 1 : x0;
156  y0 = y0 >= m_dim ? m_dim - 1 : y0;
157  z0 = z0 >= m_dim ? m_dim - 1 : z0;
158 
159  x0 = x0 < 0 ? 0 : x0;
160  y0 = y0 < 0 ? 0 : y0;
161  z0 = z0 < 0 ? 0 : z0;
162 
163  // Get other set of coords
164  unsigned int x1 = x0 >= m_dim -1 ? x0 : x0 + 1;
165  unsigned int y1 = y0 >= m_dim -1 ? y0 : y0 + 1;
166  unsigned int z1 = z0 >= m_dim -1 ? z0 : z0 + 1;
167 
168 
169  // Get values
170  T val0 = m_data[Hash(x0, y0, z0)]; //bottom font left
171  T val1 = m_data[Hash(x1, y0, z0)]; //bottom front right
172  T val2 = m_data[Hash(x0, y1, z0)]; //bottom back left
173  T val3 = m_data[Hash(x1, y1, z0)]; //bottom back right
174 
175  T val4 = m_data[Hash(x0, y0, z1)];
176  T val5 = m_data[Hash(x1, y0, z1)];
177  T val6 = m_data[Hash(x0, y1, z1)];
178  T val7 = m_data[Hash(x1, y1, z1)];
179 
180 
181  // do interpolation here
182  float dx = x - x0;
183  T valX0 = LinearInterpolate(val0, val1, dx);
184  T valX1 = LinearInterpolate(val2, val3, dx);
185  T valX2 = LinearInterpolate(val4, val5, dx);
186  T valX3 = LinearInterpolate(val6, val7, dx);
187 
188  float dy = y - y0;
189  T valY0 = LinearInterpolate(valX0, valX1, dy);
190  T valY1 = LinearInterpolate(valX2, valX3, dy);
191 
192  float dz = z - z0;
193  T valZ0 = LinearInterpolate(valY0, valY1, dz);
194 
195 
196 
197  return valZ0;
198 }
